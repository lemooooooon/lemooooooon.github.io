统计字符串中每个字符的出现频率，返回一个 Object，key 为统计字符，value 为出现频率

2020-08-03
function count(str) {
    let obj={};
    for(let i =0;i<str.length;i++){
        if(str[i]!=='\0'){
            let key=str[i];
            if(obj[key]!==undefined){
                obj[key]++;
            }
            else{
                obj[key]=1;
            }
        }
    }
    return obj;
}
